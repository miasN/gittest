dir - показывает содержимое каталога
cd .. - переходит на каталог выше
cd 'подкаталог' - перейти в подкаталог
git clone __ссылка на репозиторий__

git status
git add . или git add "_имя файла_"
git commit
git push (отправляет комит на гитхаб)

alias - в файле .gitconfig пользователя можно прописать алиасы (сокращения команд
    это упростит и ускорит работу с гит)
